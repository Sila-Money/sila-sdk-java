 *
 * @author Karlo Lorenzana

    /**
     *
     */
    private static final long serialVersionUID = 4559785420326242558L;

    /**
     * Constructor for BadRequestException.
     *
     * @param errorMessage
     */
    public BadRequestException(String errorMessage) {
        super("Bad Request: " + errorMessage);
    }
}
